# Statement example:
# main {        -> SIMULATE
#       ...     -> *contents*
# }             -> END
#
# value           -- one numerical GPSS operand
# "parameters"    -- string with GPSS operands
# name            -- word


# Program body statement
main { 		            -> SIMULATE
    ...		            -> *contents*
}		                -> END


# Device statement
block {		            -> SEIZE block
    ...		            -> *contents*
}		                -> RELEASE block


# Expression (high-level macro)
exprName {
    ...
}

# Delay statement
delay("parameters")	    -> ADVANCE parameters


# Queue start statement
q (name)	            -> QUEUE name


# Queue exit statement
dq (name)	            -> DEPART name


# Generate statement
gen ("parameters")      -> GENERATE parameters


# Start statement
init (value)          -> START value


# Terminate statement
destroy (value)       -> TERMINATE value


# Transfer statement
goto ("parameters")     -> TRANSFER parameters


# Test statement
compare ("parameters")  -> TEST parameters


# Savevalue statement
var ("parameters")      -> SAVEVALUE parameters


# Initial statement
varinit ("parameters")  -> INITIAL parameters

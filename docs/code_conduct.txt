# Token example:
# main {        -> SIMULATE
#       ...     -> *contents*
# }             -> END

# *value -- one numerical GPSS operand
# *parameters -- tuple of GPSS operands

# Program body token

main { 		-> SIMULATE
    ...		-> *contents*
}		    -> END

# Delay token

delay(*parameters)	-> ADVANCE *parameters

# Device token

block {		-> SEIZE block
    ...		-> *contents*
}		    -> RELEASE block

# Queue start token

q name		-> QUEUE name

# Queue exit token

dq name		-> DEPART name

# Expression (high-level macro)

exprName = {
    ...
}

# Generate token
gen (*parameters) --> GENERATE *parameters

# Start token
init (*value) --> START *value

# Terminate token
destroy (*value) --> TERMINATE *value

# Transfer token
goto (*parameters) --> TRANSFER *parameters
